function analyze() { const p = document.getElementById("problem").value.toLowerCase(); let title = "", explanation = "", advice = "", reminder = ""; if (p.includes("sweating") || p.includes("hyperhidrosis")) { title = "Hyperhidrosis"; explanation = "Excess sweating due to overactive sweat glands."; advice = "Stay cool, wear breathable clothes."; reminder = "Drink water frequently."; } else if (p.includes("fever") || p.includes("temperature") || p.includes("hot")) { title = "Fever"; explanation = "Rise in body temperature due to infection."; advice = "Rest and drink fluids."; reminder = "Check temperature regularly."; } else if (p.includes("cold") || p.includes("cough") || p.includes("flu")) { title = "Cold / Cough"; explanation = "Common viral infection."; advice = "Drink warm fluids, eat vitamin C rich food."; reminder = "Avoid cold drinks and rest well."; } else if (p.includes("stomach") || p.includes("tummy") || p.includes("gas") || p.includes("vomit") || p.includes("diarrhea")) { title = "Digestive Issue"; explanation = "Problem related to digestion."; advice = "Avoid junk food, eat light meals."; reminder = "Drink warm water and rest."; } else if (p.includes("headache") || p.includes("migraine") || p.includes("ache")) { title = "Headache / Migraine"; explanation = "May be due to stress, dehydration or eyestrain."; advice = "Rest, hydrate, reduce screen time."; reminder = "Avoid bright lights."; } else { // DEFAULT SMART AI title = "General Health Observation"; explanation = `Based on your input: "${p}", it may be a mild health issue.`; advice = "Monitor symptoms, stay hydrated and rest."; reminder = "Consult a doctor if it worsens."; } // SHOW RESULTS document.getElementById("result").innerHTML = `

${title}

Explanation: ${explanation}

Advice: ${advice}

Precautions / Reminder: ${reminder}

`; showPage("resultPage"); }
